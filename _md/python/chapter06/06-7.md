```python
def is_palindrome(s):
    if list(reversed(s)) == list(s):
        return True
    return False

def is_palindrome(s):
    if s == s[::-1]:
        return True
    return False

def is_palindrome(word):
    if len(word) < 2:
        return True
    if word[0] != word[-1]:
        return False
    return is_palindrome(word[1:-1])

is_palindrome('level')
is_palindrome('world')
```
